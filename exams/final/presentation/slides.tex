	\section{Architecture}
	
	\begin{frame}{Architecture \desc{Backend}}
		\begin{itemize}
			\item MVC architecture provided trough ASP.Net Core
			\item Service layer and dependency injection
			\begin{itemize}
				\item Improved testing of first party components and external dependencies
			\end{itemize}
			\item Extended modularity due to using mediator Pattern (MediatR)
			\begin{itemize}
				\item Slims down Requests into easily maintainable pieces of code
				\item Validation alongside the Request instead of the data model
			\end{itemize}
			\item Extensible due to the ability to smoothly integrate other micro-services
			\begin{itemize}
				\item Allows for re-using instead of re-implementing functionality
			\end{itemize}
		\end{itemize}
	\end{frame}
